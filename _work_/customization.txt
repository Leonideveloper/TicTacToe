class PlayerProperties:
	Id scoreDisplayId;
	Id resultDisplayId;
	IconProvider iconsProvider;
	
	def __init__(Id scoreDisplayId, Id resultDisplayId, IconProvider iconsProvider):
		self.scoreDisplayId = scoreDisplayId;
		self.resultDisplayId = resultDisplayId;
		self.iconsProvider = iconsProvider;
		

		
enum PlayerOrderType:
	Player_1;
	Player_2;
	

	
class PlayerPropertiesProvider:
	FIRST_PLAYER_PROPERTIES = PlayerProperties(
		R.id.firstScoreDisplay,
		R.id.firstResultDisplay,
		Theme.Game.GameBoard.firstPlayerIconsProvider()
	};
	SECOND_PLAYER_PROPERTIES = PlayerProperties(
		R.id.secondScoreDisplay,
		R.id.secondResultDisplay,
		Theme.Game.GameBoard.secondPlayerIconsProvider()
	};

	def getPropertiesByOrderType(PlayerOrderType orderType):
		if (orderType == PlayerOrderType.Player_1):
			return FIRST_PLAYER_PROPERTIES;
		elif (orderType == PlayerOrderType.Player_2):
			return SECOND_PLAYER_PROPERTIES;
